037    package org.jblas.util;
038    
039    import java.io.*;
040    
041    /**
042     * Class which allows to load a dynamic file as resource (for example, from a 
043     * jar-file)
044     */
045    public class LibraryLoader {
046            /** Find the library <tt>libname</tt> as a resource, copy it to a tempfile
047             * and load it using System.load(). The name of the library has to be the
048             * base name, it is mapped to the corresponding system name using 
049             * System.mapLibraryName(). For example, the library "foo" is called "libfoo.so"
050             * under Linux and "foo.dll" under Windows, but you just have to pass "foo" 
051             * the loadLibrary().
052             * 
053             * I'm not quite sure if this doesn't open all kinds of security holes. Any ideas?
054             * 
055             * @param libname basename of the library
056             */ 
057            public void loadLibrary(String libname) {
058                    libname = System.mapLibraryName(libname);
059    
060                    // We're in a static initializer and need a class. What shall we do?
061                    Class cl = getClass();
062    
063                    InputStream is = cl.getResourceAsStream("/" + libname);
064            if (is != null) {
065                System.err.println("Trying to copy from /" + libname + ".");
066            }
067            else {
068                            is = cl.getResourceAsStream("/bin/" + libname);
069                if (is != null) {
070                    System.err.println("Trying to copy from /bin/" + libname + ".");
071                }
072                else {
073                    is = cl.getResourceAsStream(fatJarLibraryPath(libname));
074                    System.err.println("Trying to copy from " + fatJarLibraryPath(libname) + ".");
075                }
076            }
077    
078                    if (is == null) {
079                            System.err.println("Couldn't find the resource " + libname + ".");
080                            System.exit(0);
081                    }
082    
083                    try {
084                            File tempfile = File.createTempFile("jblas", libname);
085                            tempfile.deleteOnExit();
086                            OutputStream os = new FileOutputStream(tempfile);
087    
088                            System.out.println("tempfile.getPath() = " + tempfile.getPath());
089    
090                            long savedTime = System.currentTimeMillis();
091    
092                            byte buf[] = new byte[1024];
093                            int len;
094                            while ((len = is.read(buf)) > 0) {
095                                    os.write(buf, 0, len);
096                            }
097    
098                            double seconds = (double) (System.currentTimeMillis() - savedTime) / 1e3;
099                            System.err.println("Copying took " + seconds + " seconds.");
100    
101                            os.close();
102    
103                            System.load(tempfile.getPath());
104                    } catch (IOException io) {
105                            System.err.println("Could not create the temp file: " + io.toString() + ".\n");
106                    } catch (UnsatisfiedLinkError ule) {
107                            System.err.println("Couldn't load copied link file: " + ule.toString() + ".\n");
108                    }
109            }
110    
111        static public String unifyOSName(String osname) {
112            if (osname.startsWith("Windows")) {
113                return "Windows";
114            }
115            return osname;
116        }
117    
118        /** Compute the path to the library. The path is basically
119            "/" + os.name + "/" + os.arch + "/" + libname. */
120        static public String fatJarLibraryPath(String libname) {
121            String sep = "/"; //System.getProperty("file.separator");
122            String os_name = unifyOSName(System.getProperty("os.name"));
123            String os_arch = System.getProperty("os.arch");
124            return sep + "lib" + sep + os_name + sep + os_arch + sep + libname;
125        }
126    }
